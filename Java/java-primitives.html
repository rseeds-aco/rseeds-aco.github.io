byte
1 byte · 8 bits Integer
Value range
−128  to  127
8 bits
byte age = 25;
byte minVal = -128;
byte maxVal = 127;
// Compact; useful for streams & arrays
short
2 bytes · 16 bits Integer
Value range
−32,768  to  32,767
16 bits
short year = 2024;
short temp = -273;
short score = 32_767;
// Rare; saves memory in large arrays
int
4 bytes · 32 bits Integer
Value range
−2,147,483,648  to  2,147,483,647
32 bits (shown × ½)
int count = 0;
int population = 8_100_000_000; // ⚠ overflow!
int hex = 0xFF_A0;
// Default integer type in Java
long
8 bytes · 64 bits Integer
Value range
−9.2 × 10¹⁸  to  9.2 × 10¹⁸
64 bits (shown × ¼)
long worldPop = 8_100_000_000L;
long timestamp = System.currentTimeMillis();
long binary = 0b1010_1111L;
// Suffix L required for long literals
float
4 bytes · 32 bits Floating-Point
Approximate range
±1.4 × 10⁻⁴⁵  to  ±3.4 × 10³⁸  (~7 sig. digits)
IEEE 754
float pi = 3.14159f;
float gravity = 9.81f;
float sci = 1.5e-3f;
// Suffix f required for float literals
double
8 bytes · 64 bits Floating-Point
Approximate range
±5 × 10⁻³²⁴  to  ±1.8 × 10³⁰⁸  (~15–16 sig. digits)
IEEE 754 double
double pi = 3.141592653589793;
double price = 19.99;
double inf = 1.0 / 0.0; // Infinity
// Default decimal type in Java
char
2 bytes · 16 bits Character
Value range
\u0000 (0)  to  \uFFFF (65,535) — Unicode BMP
Unsigned 16-bit
char letter = 'A';
char newline = '\n';
char euro = '\u20AC'; // €
// Single quotes only; unsigned type
boolean
JVM-dependent* Boolean
Valid values
true  or  false  — only two possible values
1 logical bit (stored as int by JVM)
boolean isActive = true;
boolean hasError = false;
boolean check = score > 90;
// Cannot cast to/from numeric types
Quick notes:   All primitives are stored by value (not reference).  ·  Use L suffix for long literals and f suffix for float literals.  ·  double is the default decimal type; int is the default integer type.  ·  * boolean size is not precisely specified by the JLS; the JVM typically uses 32 bits for a single boolean variable and packs boolean arrays as 1 bit per element.
